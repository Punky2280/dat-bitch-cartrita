# OpenAI project / standard keys (sk- or sk-proj-)
regex:sk-[A-Za-z0-9_-]{20,}==>sk-REDACTED
regex:sk-proj-[A-Za-z0-9_-]{20,}==>sk-proj-REDACTED

# GitHub Personal Access Tokens (classic & fine-grained usually start ghp_)
regex:ghp_[A-Za-z0-9]{30,}==>ghp_REDACTED

# GitLab Personal Access Tokens (glpat- prefix)
regex:glpat-[A-Za-z0-9_-]{20,}==>glpat-REDACTED

# LangSmith / LangChain keys (lsv2_ prefix)
regex:lsv2_[A-Za-z0-9_]{10,}==>lsv2_REDACTED

# Deepgram
regex:[0-9a-f]{32}==>hex32_REDACTED

# Google API Key (AIzaSy pattern)
regex:AIza[0-9A-Za-z\-_]{20,}==>AIzaSy_REDACTED

# Tavily
regex:tvly-[A-Za-z0-9_-]{10,}==>tvly-REDACTED

# SerpAPI
regex:[0-9a-f]{64}==>sha256_REDACTED

# Wolfram Alpha (short tokens; optional, broad pattern)
regex:[A-Z0-9]{10}==>TOKEN10_REDACTED

# Hugging Face
regex:hf_[A-Za-z0-9]{30,}==>hf_REDACTED

# Encryption key (64+ hex chars) - ensure not to obliterate legitimate hashes universally
regex:[0-9a-f]{64}==>hex64_REDACTED

# JWT secrets (heuristic: 'secret' + any word chars)
regex:secret-[A-Za-z0-9_-]{8,}==>secret-REDACTED
